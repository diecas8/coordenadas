    <script>

        function convertToUTM() {

            const lat = parseFloat(document.getElementById('lat').value);

            const lon = parseFloat(document.getElementById('lon').value);

 

            if (isNaN(lat) || isNaN(lon)) {

                document.getElementById('output').innerText = 'Please enter valid geographic coordinates.';

                return;

            }

 

            // Define WGS84 and UTM projection

            const wgs84 = "EPSG:4326";

            const utm = `+proj=utm +zone=${Math.floor((lon + 180) / 6) + 1} +datum=WGS84 +units=m +no_defs`;

 

            // Convert

            const [easting, northing] = proj4(wgs84, utm, [lon, lat]);

 

            document.getElementById('output').innerText = `UTM Coordinates:\nEasting: ${easting.toFixed(2)}\nNorthing: ${northing.toFixed(2)}`;

        }

 

        function convertToGeographic() {

            const easting = parseFloat(document.getElementById('easting').value);

            const northing = parseFloat(document.getElementById('northing').value);

            const zone = parseInt(document.getElementById('zone').value);

 

            if (isNaN(easting) || isNaN(northing) || isNaN(zone)) {

                document.getElementById('output').innerText = 'Please enter valid UTM coordinates and zone.';

                return;

            }

 

            // Define WGS84 and UTM projection

            const utm = `+proj=utm +zone=${zone} +datum=WGS84 +units=m +no_defs`;

            const wgs84 = "EPSG:4326";

 

            // Convert

            const [lon, lat] = proj4(utm, wgs84, [easting, northing]);

 

            document.getElementById('output').innerText = `Geographic Coordinates:\nLatitude: ${lat.toFixed(6)}\nLongitude: ${lon.toFixed(6)}`;

        }

    </script>
